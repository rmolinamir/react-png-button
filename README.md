# react-png-button

## Install

```bash
npm install --save react-png-button
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'react-png-button'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
